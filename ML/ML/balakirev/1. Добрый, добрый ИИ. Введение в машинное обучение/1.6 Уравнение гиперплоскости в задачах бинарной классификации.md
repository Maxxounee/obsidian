```python
import numpy as np

x_test = [np.array([1, x1, x2]) for x1, x2 in [(5, -3), (-3, 8), (3, 6), (0, 0), (5, 3), (-3, -1), (-3, 3)]]

b = w0 = 33 / 13
k = w1 = -9 / 13
w2 = -1

w = np.array([w0, w1, w2])

def get_class(w, x):
    return 1 if w.T @ x < 0 else -1

predict = [get_class(w, x) for x in x_test]
```

```python
import numpy as np

x_test = [np.array([1, x1, x2]) for x1, x2 in [(9, 6), (2, 4), (-3, -1), (3, -2), (-3, 6), (7, -3), (6, 2)]]

point = [7, 7]
b = w0 = -14 / 5
k = w1 = 7 / 5
w2 = (-w1 * point[0] - w0) / point[1]

w = np.array([w0, w1, w2])

def get_class(w, x):
    return 1 if w.T @ x < 0 else -1

predict = [get_class(w, x) for x in x_test]
```
